\chapter{The triangulation theorem: the Moise route}

This chapter records the formalization of Rad\'o's theorem (every compact
bordered surface admits a finite triangulation) following Moise,
\emph{Geometric Topology in Dimensions 2 and 3}, Ch.~8, whose inputs are the
\emph{polygonal} Jordan and Schoenflies theorems (Ch.~2--3) and the PL
approximation theorem (Ch.~5--6).  The full Jordan curve theorem (Ch.~4) and
the full Schoenflies theorem (Ch.~9) are not on this route.  The Lean files
live in \texttt{ClassificationOfSurfaces/Moise/}.

\section{The Eval surface interface}

\begin{definition}[C0 chart-boundary invariance]
\label{moise:def:chart_boundary_invariant}
\lean{LeanEval.Topology.ClassificationOfSurfaces.ChartBoundaryInvariant}
\leanok
The one deliberate assumption of the project: a mathlib manifold-boundary point
lying in any preferred half-plane chart is sent by that chart to the frontier of
the chart's extended target.  Mathlib proves this for positive regularity
(\texttt{ModelWithCorners.isBoundaryPoint\_iff\_of\_mem\_atlas}); at regularity
$C^0$ it is the invariance-of-boundary theorem, deferred as this typeclass.
Only this implication is assumed; its converse is never used.
\end{definition}

\begin{definition}[Lean Eval compact surface with boundary]
\label{moise:def:eval_surface}
\lean{LeanEval.Topology.ClassificationOfSurfaces.EvalSurface}
\leanok
\uses{moise:def:chart_boundary_invariant}
An \emph{Eval surface} is a Hausdorff, connected, compact type charted over the
Euclidean half-plane, a $C^0$ manifold for
\texttt{modelWithCornersEuclideanHalfSpace 2}, together with the
\ref{moise:def:chart_boundary_invariant} bridge.
\end{definition}

\section{Geometric triangulations}

\begin{definition}[Geometric realization of a face family]
\label{moise:def:geometric_realization}
\lean{LeanEval.Topology.ClassificationOfSurfaces.GeometricRealization}
\leanok
For a finite vertex type $V$ and a family $F$ of three-element vertex sets, the
realization is the set of points of the standard simplex
$\Delta^V\subseteq V\to\mathbb R$ supported on some face of $F$: the finite
union of the geometric $2$-simplexes of $F$, glued along shared barycentric
faces.  It is compact and Hausdorff by construction.
\end{definition}

\begin{definition}[Geometric triangulation]
\label{moise:def:geometric_triangulation}
\lean{LeanEval.Topology.ClassificationOfSurfaces.GeometricTriangulation}
\leanok
\uses{moise:def:geometric_realization}
A \emph{geometric triangulation} of a space $S$ is a finite face family
together with a homeomorphism from its geometric realization onto $S$.  There
is no independent realization field: the realization is computed from the
combinatorial data, so the definition cannot be satisfied by degenerate
witnesses.  Executable anchors: the standard $2$-simplex triangulates itself,
and $\mathbb R$ and $\mathbb Q$ admit no geometric triangulation
(\texttt{Moise/Countermodels.lean}).
\end{definition}

\begin{lemma}[Triangulated spaces are compact Hausdorff]
\label{moise:lem:triangulation_compact}
\lean{LeanEval.Topology.ClassificationOfSurfaces.GeometricTriangulation.compactSpace, LeanEval.Topology.ClassificationOfSurfaces.GeometricTriangulation.t2Space}
\leanok
\uses{moise:def:geometric_triangulation}
A space with a geometric triangulation is compact and Hausdorff.
\end{lemma}

\begin{proof}
\leanok
The realization is a closed subset of the compact standard simplex inside the
Hausdorff space $V\to\mathbb R$; transport along the homeomorphism.
\end{proof}

\begin{theorem}[Rad\'o; Moise Ch.~8, Thm.~3]
\label{moise:thm:moise_triangulation}
\lean{LeanEval.Topology.ClassificationOfSurfaces.moise_triangulation}
\leanok
\uses{moise:def:eval_surface,moise:def:geometric_triangulation,moise:thm:rado_assembly}
Every Eval surface admits a geometric triangulation.
\end{theorem}

\begin{proof}
\leanok
This is \ref{moise:thm:rado_assembly}; the remaining mathematical content
lives on that theorem's open inputs.
\end{proof}

\section{Plane complexes and piecewise-linear maps}

\begin{definition}[Finite simplicial complex in the plane]
\label{moise:def:plane_complex}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PlaneComplex}
\leanok
A \emph{plane complex} is a finite family of affine simplexes of dimension at
most two in $\mathbb R^2$: vertex positions are actual points, each face is
affinely independent, faces are closed under nonempty subsets, and any two face
carriers (convex hulls) intersect exactly in the carrier of their shared
vertices.  The support is defined as the union of the face carriers and is
compact.  Executable anchor: the closed standard triangle with its seven faces
(\texttt{Moise/Anchors.lean}).
\end{definition}

\begin{definition}[Subdivisions and PL maps]
\label{moise:def:pl_maps}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PlaneComplex.Subdivides, LeanEval.Topology.ClassificationOfSurfaces.Moise.IsAffineOn, LeanEval.Topology.ClassificationOfSurfaces.Moise.IsPLOn, LeanEval.Topology.ClassificationOfSurfaces.Moise.IsPLOnSet}
\leanok
\uses{moise:def:plane_complex}
$K'$ subdivides $K$ when they have the same support and every face carrier of
$K'$ lies in a face carrier of $K$.  A map is \emph{PL on $K$} when it agrees
with an affine map on every face carrier of some subdivision of $K$, and PL on
a set $A$ when $A$ carries some plane complex on which it is PL.  A generic
continuous map is not PL on any complex with a $2$-face.
\end{definition}

\begin{lemma}[Realization bridge]
\label{moise:lem:plane_complex_realization}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PlaneComplex.toGeometricTriangulation}
\leanok
\uses{moise:def:plane_complex,moise:def:geometric_triangulation}
A purely two-dimensional plane complex induces a geometric triangulation of its
support, by barycentric coordinates in the face containing each point.
\end{lemma}

\begin{proof}
\leanok
Elementary affine geometry and uniqueness of barycentric coordinates on
affinely independent faces, glued using the face-to-face intersection axiom.
\end{proof}

\section{The Jordan curve theorem for polygons (Moise Ch.~2)}

\begin{definition}[Polygonal simple closed curve]
\label{moise:def:polygonal_circle}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle}
\leanok
A \emph{polygonal circle} is a cyclically indexed family of at least three
plane points with consecutive points distinct, adjacent segments meeting
exactly in the shared vertex, and non-adjacent segments disjoint.  Its carrier
(the union of the edges) is compact.  Executable anchor: the standard triangle
(\texttt{Moise/Anchors.lean}).
\end{definition}

\begin{definition}[Crossing index]
\label{moise:def:crossing_index}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.EdgeCrossed, LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.index}
\leanok
\uses{moise:def:polygonal_circle}
The \emph{index} of a plane point is the parity of the number of polygon edges
crossed by its leftward horizontal ray, an edge counting as crossed when the
point's height lies in the half-open interval between the endpoint heights.
The half-open convention replaces Moise's perturbed lines: the index is
everywhere defined, no general-position choice of axes is needed, and
horizontal edges are never crossed.
\end{definition}

\begin{lemma}[The index vanishes far away]
\label{moise:lem:index_zero_high}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.index_eq_zero_of_high, LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.exists_index_eq_zero}
\leanok
\uses{moise:def:crossing_index}
A point strictly above every vertex is off the polygon and has index zero; such
points exist.
\end{lemma}

\begin{proof}
\leanok
No edge satisfies the height condition; points of a segment have heights
bounded by the endpoint heights.
\end{proof}

\begin{lemma}[Local constancy of the index]
\label{moise:lem:index_locally_constant}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.index_locallyConstant}
\leanok
\uses{moise:def:crossing_index}
Off the polygon the index is locally constant.
\end{lemma}

\begin{proof}
\leanok
Casework: near a point off the polygon, the crossing status of each edge is
unchanged unless the ray height passes a vertex height, where the half-open
convention makes the count change by $0$ or $2$.
\end{proof}

\begin{lemma}[An index-one point exists]
\label{moise:lem:index_one_exists}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.exists_index_eq_one}
\leanok
\uses{moise:def:crossing_index}
Some point off the polygon has index one.
\end{lemma}

\begin{proof}
\leanok
Choose a non-vertex height attained by the polygon, take the leftmost polygon
point at that height, and move slightly right of it (Moise's construction;
requires that the polygon is not contained in a single horizontal line, which
follows from the embedding fields).
\end{proof}

\begin{lemma}[Moise Ch.~2, Thm.~1, Lemma 2]
\label{moise:lem:complement_disconnected}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.compl_carrier_not_isPreconnected}
\leanok
\uses{moise:lem:index_zero_high,moise:lem:index_locally_constant,moise:lem:index_one_exists}
The complement of a polygon is disconnected.
\end{lemma}

\begin{proof}
\leanok
The index-$0$ and index-$1$ loci are relatively open by local constancy, cover
the complement, and are both nonempty.
\end{proof}

\begin{theorem}[Jordan curve theorem for polygons; Moise Ch.~2, Thms.~1, 5, 6]
\label{moise:thm:polygonal_jordan}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.polygonal_jordan}
\leanok
\uses{moise:def:polygonal_circle,moise:lem:complement_disconnected}
The complement of a polygon has exactly two connected components, an open
bounded interior and an open unbounded exterior, and the polygon is the
frontier of each.
\end{theorem}

\begin{proof}
\leanok
The crossing-index argument, boundedness distinction, visibility segment to
the strip, and the global assembly from a two-gate strip are formalized.  The
explicit strip is also constructed: quantitative vertex balls and isolated
central edge rectangles cover the polygon, and its punctured part decomposes
into local sectors and half-rectangles.  Explicit rotated points prove the
four endpoint overlaps for each edge.  A finite cyclic graph argument makes
the two resulting bands path-connected, while closure of the parameter
rectangles proves that both bands approach every point of the polygon.  These
data instantiate \texttt{TwoGateStrip}; the crossing index and the
visibility-to-strip lemma then give exactly two complementary components,
their bounded/unbounded distinction, and both frontier equalities.
\end{proof}

\begin{definition}[Interior, exterior, closed region]
\label{moise:def:polygon_regions}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.interiorRegion, LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.exteriorRegion, LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.closedRegion}
\leanok
\uses{moise:thm:polygonal_jordan}
The interior and exterior regions of a polygon are the bounded and unbounded
complementary components provided by \ref{moise:thm:polygonal_jordan}; the
closed region is the closure of the interior, and it is compact.
\end{definition}

\section{The Schoenflies theorem for polygons (Moise Ch.~2--3)}

\begin{theorem}[Polygonal disks are polyhedra; Moise Ch.~2, Thm.~2]
\label{moise:thm:region_polyhedron}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.closedRegion_is_polyhedron}
\leanok
\uses{moise:def:polygon_regions,moise:def:plane_complex}
The closed region bounded by a polygon is the support of a finite, purely
two-dimensional plane complex.
\end{theorem}

\begin{proof}
Place the compact closed region in a large triangle, subdivide that triangle
successively by the finitely many affine lines through the polygon's edges, and
retain the chambers on the bounded side of polygonal Jordan.  Each chamber
interior misses the polygon because it lies on one closed side of every cutting
line.  The resulting finite submesh has support exactly the closed region.
\end{proof}

\begin{lemma}[Polygonal disks are triangulable]
\label{moise:lem:region_triangulable}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.closedRegion_triangulable}
\leanok
\uses{moise:thm:region_polyhedron,moise:lem:plane_complex_realization}
The closed region bounded by a polygon admits a geometric triangulation.
\end{lemma}

\begin{proof}
\leanok
Combine \ref{moise:thm:region_polyhedron} with the realization bridge.
\end{proof}

\begin{theorem}[Schoenflies for polygons; Moise Ch.~3, Thm.~5]
\label{moise:thm:polygonal_schoenflies}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.polygonal_schoenflies}
\uses{moise:thm:polygonal_schoenflies_rel}
Any two polygons in the plane are equivalent under a homeomorphism of the whole
plane.
\end{theorem}

\begin{proof}
Apply relative straightening to both polygons and compose the resulting maps
with the affine ambient equivalence between the two target triangles.
\end{proof}

\begin{theorem}[Relative Schoenflies for polygons; Moise Ch.~3, Thm.~7]
\label{moise:thm:polygonal_schoenflies_rel}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PolygonalCircle.polygonal_schoenflies_rel}
\uses{moise:thm:region_polyhedron}
The straightening homeomorphism can be chosen supported in any prescribed open
set containing the closed region: it sends the polygon to the frontier of a
triangle and is the identity outside the open set.
\end{theorem}

\begin{proof}
Choose the homeomorphisms in Moise's Thm.~4 induction supported in the open
set.
\end{proof}

\section{Broken lines (Moise Ch.~1)}

\begin{definition}[Broken-line joinedness]
\label{moise:def:broken_line}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.JoinedByBrokenLine}
\leanok
Two points of a set are joined by a broken line when a finite chain of segments
inside the set connects them.
\end{definition}

\begin{lemma}[Open connected sets are broken-line connected]
\label{moise:lem:broken_line_connected}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.IsPreconnected.joinedByBrokenLine}
\leanok
\uses{moise:def:broken_line}
In an open preconnected subset of the plane, any two points are joined by a
broken line.
\end{lemma}

\begin{proof}
\leanok
Clopen-chain argument: the locus joined to a fixed point is open (extend by one
segment inside a ball) and relatively closed (same ball argument), hence
everything.
\end{proof}

\section{PL approximation (Moise Ch.~5--6)}

\begin{theorem}[Cone extension; Moise Ch.~5, Thms.~3--6]
\label{moise:thm:cone_extension}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.pl_extension_of_triangle_boundary}
\leanok
\uses{moise:def:pl_maps,moise:thm:polygonal_jordan}
A map that is PL and injective on the frontier of a triangle, carrying it onto
the frontier of a second triangle, extends to a PL homeomorphism of the closed
triangles.
\end{theorem}

\begin{proof}
Cone from an interior point: the extension is affine on each segment from the
cone point to the boundary.
\end{proof}

\begin{theorem}[One-skeleton approximation; Moise Ch.~6, Thm.~2]
\label{moise:thm:one_skeleton_approx}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.pl_approximation_one_skeleton}
\leanok
\uses{moise:def:pl_maps,moise:lem:broken_line_connected}
An embedding of the support of a finite one-dimensional plane complex into the
plane can be $\varepsilon$-approximated by a PL embedding agreeing with it on
all vertices.
\end{theorem}

\begin{proof}
Subdivide finely, replace each small arc by a broken line in a small
neighborhood (\ref{moise:lem:broken_line_connected}), keeping the broken lines
disjoint except at shared endpoints.
\end{proof}

\begin{theorem}[PL approximation of embedded 2-manifolds; Moise Ch.~6, Thm.~3]
\label{moise:thm:pl_approximation}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.pl_approximation_two_manifold}
\leanok
\uses{moise:thm:one_skeleton_approx,moise:thm:cone_extension,moise:thm:polygonal_schoenflies_rel}
An embedding of the support of a finite combinatorial $2$-manifold-with-boundary
into the plane can be $\varepsilon$-approximated by a PL embedding.  This is
the single point of the whole route where dimension two is used (Moise's remark
at the end of Ch.~8).
\end{theorem}

\begin{proof}
\leanok
Approximate on the one-skeleton by \ref{moise:thm:one_skeleton_approx}, extend
across each $2$-cell by the combinatorial Schoenflies theorem
(\ref{moise:thm:cone_extension}, via \ref{moise:thm:polygonal_schoenflies_rel}),
choosing the graph perturbation below a uniform vertex--cell separation scale.
A bounded Tietze extension of the boundary displacement and no-retraction
detect the correct side of every image polygon.  Finite side propagation then
shows that distinct cell interiors are disjoint, so the cellwise extensions
glue to the required PL embedding.
\end{proof}

\section{Chart extraction (Moise Ch.~8, Thm.~1)}

\begin{definition}[Moise charts]
\label{moise:def:moise_chart}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.ChartKind, LeanEval.Topology.ClassificationOfSurfaces.Moise.MoiseChart, LeanEval.Topology.ClassificationOfSurfaces.Moise.MoiseChart.core}
\leanok
A \emph{Moise chart} is an open domain homeomorphic to the standard open unit
disk (interior kind) or half-disk (boundary kind), with a compact core marked
out by the model disk of radius one half.
\end{definition}

\begin{definition}[Boundary faithfulness]
\label{moise:def:boundary_faithful}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.MoiseChart.BoundaryFaithful}
\leanok
\uses{moise:def:moise_chart,moise:def:chart_boundary_invariant}
A chart is boundary-faithful when disk charts contain no manifold-boundary
points and half-disk charts send manifold-boundary points to the model edge
line.  This is exactly what \ref{moise:def:chart_boundary_invariant} yields;
the converse of the half-disk clause is $C^0$ invariance of the boundary and is
deliberately not part of the interface.
\end{definition}

\begin{lemma}[Local chart extraction]
\label{moise:lem:chart_extraction}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.exists_moiseChart_core_mem_nhds}
\leanok
\uses{moise:def:eval_surface,moise:def:moise_chart,moise:def:boundary_faithful}
Every point of an Eval surface has a boundary-faithful Moise chart whose core
is a neighborhood of the point.
\end{lemma}

\begin{proof}
\leanok
Take the preferred mathlib chart; a small ball (interior case) or relative
half-ball (edge case) about the image point lies in the chart image, and the
recentering homeomorphism $v\mapsto\varepsilon^{-1}(v-p)$ straightens it onto
the standard model.  \ref{moise:def:chart_boundary_invariant} places boundary
points on the frontier of the extended target, hence on the edge line.
\end{proof}

\begin{lemma}[Finite chart cover]
\label{moise:lem:finite_chart_cover}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.moise_finite_chart_cover}
\leanok
\uses{moise:lem:chart_extraction}
A compact Eval surface has a finite cover by boundary-faithful Moise chart
cores.
\end{lemma}

\begin{proof}
\leanok
Finite subcover of the core interiors.
\end{proof}

\section{The Rad\'o induction (Moise Ch.~8, Thm.~3)}

\begin{definition}[Partial triangulation]
\label{moise:def:partial_triangulation}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PartialTriangulation, LeanEval.Topology.ClassificationOfSurfaces.Moise.PartialTriangulation.combInterior}
\leanok
\uses{moise:def:geometric_realization}
A \emph{partial triangulation} of $S$ is a finite face family together with a
topological embedding of its geometric realization into $S$.  Its support is
compact.  Its combinatorial interior is the support minus the image of the
boundary edges (edges lying in exactly one face); for the bordered induction,
the ambient topological interior of the support is the relevant neighborhood
notion.
The fixed disk/half-disk diamond patch transports through any Moise chart to
an explicit partial triangulation whose ambient interior contains the chart
core; it satisfies the Rad\'o invariant for that single core.
\end{definition}

\begin{definition}[The Rad\'o invariant]
\label{moise:def:rado_invariant}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.RadoInvariant}
\leanok
\uses{moise:def:partial_triangulation}
The invariant carried through the bordered induction: the complex is a
combinatorial surface (every edge in at most two faces) and the absorbed region
lies in the topological interior of the support as a subset of the ambient
surface.  This includes genuine boundary points of half-disk cores, unlike
combinatorial interior.  The empty partial triangulation satisfies the
invariant for the empty region.  This invariant is expected to be strengthened during the proof of
\ref{moise:thm:induction_step}; weakening the step's conclusion is not
permitted.
\end{definition}

\begin{lemma}[Faithful intrinsic fine subdivision]
\label{moise:lem:intrinsic_fine_subdivision}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.IntrinsicTwoComplex.exists_subdivision_mesh_lt}
\leanok
\uses{moise:def:partial_triangulation}
Every finite intrinsic two-complex has a faithful subdivision of arbitrarily
small barycentric mesh.  Iterated one-to-four midpoint subdivision supplies
the subdivision and divides the mesh bound by two at every step.
\end{lemma}

\begin{proof}
\leanok
Construct the midpoint complex with old vertices and one vertex for each old
edge.  Explicit barycentric recovery proves that its affine evaluation is a
homeomorphism onto the old realization.  Corner and central triangles have at
most half the old diameter; iterate.
\end{proof}

\begin{lemma}[Finite compact collar]
\label{moise:lem:finite_compact_collar}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PartialTriangulation.exists_refinedSubcomplex_between}
\leanok
\uses{moise:lem:intrinsic_fine_subdivision}
If a compact subset of a partial triangulation lies in an ambient open set,
then it is carried by a finite face restriction of a faithful refinement whose
support still lies in that open set.
\end{lemma}

\begin{proof}
\leanok
Apply the Lebesgue-number lemma to the two-set cover consisting of the open set
and the complement of the compact set, then retain exactly the refined faces
subordinate to the open set.
\end{proof}

\begin{lemma}[Vanishing-error frontier glue]
\label{moise:lem:frontier_glue}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.continuous_frontierGlue}
\leanok
If a continuous map is replaced on an open set by a continuous approximation
whose error tends to zero at the frontier, the replacement and the unchanged
map glue to a continuous global map.  Distance to the complement is strongly
positive on compact subsets of the open set and gives a canonical
frontier-vanishing control.
\end{lemma}

\begin{proof}
\leanok
At interior and exterior points continuity is local.  At a frontier point,
combine the approximation bound with continuity of the unchanged map and the
triangle inequality.
\end{proof}

\begin{theorem}[Gluing along a common presentation; Moise Thm.~7.6]
\label{moise:thm:partial_glue}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.PartialTriangulation.exists_glued}
\leanok
\uses{moise:def:geometric_realization}
Two partial triangulations presented on a common vertex type, whose embeddings
agree exactly on the shared part of their realizations, glue to a single
partial triangulation on the union face family, supported on the union of the
two images, with the joint edge-face bound inherited.
\end{theorem}

\begin{proof}
\leanok
The realization of the union family is the set-union of the two realizations,
so the glued embedding is the two-sided paste; it is injective by the exact
agreement hypotheses, continuous because each side's preimage of a closed set
is a compact inclusion image, and an embedding as a continuous injection from
a compact realization into a Hausdorff surface.
\end{proof}

\begin{theorem}[Crossing-case welding; Moise Ch.~8, Thm.~3 preparation]
\label{moise:thm:crossing_weld}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.MoiseChart.exists_crossing_weld}
\leanok
\uses{moise:def:rado_invariant,moise:def:boundary_faithful,moise:thm:pl_approximation,moise:lem:finite_compact_collar,moise:lem:frontier_glue}
In the genuine crossing case, the straightened old complex and the chart patch
admit a common welded presentation: a common vertex type carrying both face
families, embeddings agreeing exactly on the shared realization, the joint
edge-face bound, and the united image containing the absorbed region and the
new core in its topological interior.
\end{theorem}

\begin{proof}
Straighten the old complex over the chart overlap by the locally finite
controlled polygonal replacement with tolerance vanishing at the overlap
frontier, refine the straightened trace and the fixed patch complex to a
common plane subdivision (Moise's conditions (a)--(h)), and read off the
welded presentation.  The finite compact-collar theorem must not be silently
substituted for the vanishing-tolerance construction.
\end{proof}

\begin{theorem}[The induction step; Moise Ch.~8, Thm.~3, step]
\label{moise:thm:induction_step}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.moise_induction_step}
\leanok
\uses{moise:def:rado_invariant,moise:def:boundary_faithful,moise:thm:partial_glue,moise:thm:crossing_weld}
Given a partial triangulation satisfying the invariant for a region $A$ and one
more boundary-faithful chart, some partial triangulation satisfies the
invariant for $A$ together with the chart's core.
\end{theorem}

\begin{proof}
\leanok
If the new core already lies in the interior of the built support, absorb it
directly; if the absorbed region lies inside the chart patch, replace the
built complex by the patch.  In the remaining crossing case, apply the welding
theorem and glue the two presentations by \ref{moise:thm:partial_glue}; the
invariant for the union follows from the welding conclusions.
\end{proof}

\begin{theorem}[Assembled Rad\'o induction]
\label{moise:thm:rado_assembly}
\lean{LeanEval.Topology.ClassificationOfSurfaces.Moise.moise_triangulation_of_boundaries}
\leanok
\uses{moise:lem:finite_chart_cover,moise:thm:induction_step,moise:def:rado_invariant,moise:def:geometric_triangulation}
A compact Eval surface admits a geometric triangulation.
\end{theorem}

\begin{proof}
\leanok
Starting from the empty complex, absorb the finitely many chart cores one at a
time by \ref{moise:thm:induction_step}; when all cores are absorbed the
invariant forces the support to be everything, and a fully covering partial
triangulation is a geometric triangulation.
\end{proof}
